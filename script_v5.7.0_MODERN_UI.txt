        // Search functionality
        document.getElementById('searchInput').addEventListener('keyup', renderList);

        function renderList() {
            const listContainer = document.getElementById('listContainer');
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            let items = [];

            if (currentTab === 'satellites') {
                items = data.satellites.filter(s => 
                    s.name.toLowerCase().includes(searchTerm) || 
                    s.constellation.toLowerCase().includes(searchTerm)
                );
            } else {
                items = data.sensors.filter(s => 
                    s.name.toLowerCase().includes(searchTerm) || 
                    s.type.toLowerCase().includes(searchTerm)
                );
            }

            listContainer.innerHTML = items.map(item => {
                const key = currentTab === 'satellites' ? 'norad_id' : 'sensor_id';
                const displayText = currentTab === 'satellites' ? 
                    `${item.name} (${item.constellation})` : 
                    `${item.name} (${item.type})`;
                
                return `<div class="list-item" onclick="selectItem('${key}', ${item[key]}, '${currentTab}')">${displayText}</div>`;
            }).join('');

            if (items.length === 0) {
                listContainer.innerHTML = '<div class="no-data">No results found</div>';
            }
        }

        function selectItem(key, value, type) {
            selectedItem = { type, key, value };
            
            // Update active state
            document.querySelectorAll('.list-item').forEach(item => item.classList.remove('active'));
            event.target.classList.add('active');

            displayDetails();
            displayRelationships();

        function displayRelationships() {
            const relationshipsList = document.getElementById('relationshipsList');
            const type = selectedItem.type;
            const value = selectedItem.value;

            let relationships = [];

            if (type === 'satellites') {
                const satLinks = data.satellite_sensor_links.filter(link => link.norad_id === value);
                relationships = satLinks.map(link => {
                    const sensor = data.sensors.find(s => s.sensor_id === link.sensor_id);
                    return { 
                        name: sensor.name, 
                        type: sensor.type, 
                        id: sensor.sensor_id,
                        relType: 'sensor',
                        clickType: 'sensors'
                    };
                });
            } else {
                const sensorLinks = data.satellite_sensor_links.filter(link => link.sensor_id === value);
                relationships = sensorLinks.map(link => {
                    const sat = data.satellites.find(s => s.norad_id === link.norad_id);
                    return { 
                        name: sat.name, 
                        type: sat.constellation, 
                        id: sat.norad_id,
                        relType: 'satellite',
                        clickType: 'satellites'
                    };
                });
            }

            if (relationships.length === 0) {
                relationshipsList.innerHTML = '<div class="no-data">No relationships found</div>';
                return;
            }

            relationshipsList.innerHTML = `<div style="color: var(--color-text-secondary); font-size: 12px; margin-bottom: 10px;">${relationships.length} linked ${type === 'satellites' ? 'sensors' : 'satellites'}</div>` +
                relationships.map(rel => 
                    `<div class="relationship-item ${rel.relType === 'sensor' ? 'sensor-type' : ''}" 
                        onclick="switchAndSelect('${rel.clickType}', ${rel.id})">
                        <strong>${rel.name}</strong><br/>
                        <small style="color: var(--color-text-secondary);">${rel.type}</small>
                    </div>`
                ).join('');
        }

        function switchAndSelect(tabName, itemId) {
            // Switch tab
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.querySelector(`[data-tab="${tabName}"]`).classList.add('active');
            currentTab = tabName;
            
            // Render list and auto-select
            renderList();
            setTimeout(() => {
                const itemToSelect = currentTab === 'satellites' ? 
                    data.satellites.find(s => s.norad_id === itemId) :
                    data.sensors.find(s => s.sensor_id === itemId);
                
                const items = document.querySelectorAll('.list-item');
                const displayText = currentTab === 'satellites' ? 
                    `${itemToSelect.name} (${itemToSelect.constellation})` :
                    `${itemToSelect.name} (${itemToSelect.type})`;
                
                const targetItem = Array.from(items).find(item => item.textContent.includes(displayText));
                if (targetItem) {
                    targetItem.click();
                }
            }, 0);
        }

        // Initial render
        renderList();
